class MinStack {
    class Node{
        int min;
        int data;
        Node next;
        Node(int data){
            this.data = data;
        }
    }
    Node head;
    public MinStack() {
        head = null;
    }
    
    public void push(int val) {
        if(head == null){
            Node newNode = new Node(val);
            newNode.min = val;
            head = newNode;
        }
        else{
            Node newNode = new Node(val);
            newNode.min = Math.min(val , head.min);
            newNode.next = head;
            head = newNode;
        }
    }
    
    public void pop() {
        if(head != null)
            head = head.next;
    }
    
    public int top() {
        if(head != null){
            return head.data;
        }
        else{
            return Integer.MIN_VALUE;
        }
    }
    
    public int getMin() {
        if(head == null){
            return Integer.MIN_VALUE;
        }
        return head.min;
    }
}
